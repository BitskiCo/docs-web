---
title: "GET Token Templates"
api: "GET https://api.bitski.com/v1/apps/{appId}/token-templates"
authMethod: "bearer"
description: "List token templates"
---

<Note>This API requires the `apps` scope</Note>

### Path

<ParamField path="appId" type="string" required>
  Your Bitski Application ID
</ParamField>

### Parameters

<ParamField query="limit" type="number" optional>
  The total number of results to include per page.
</ParamField>

<ParamField query="offset" type="number" optional>
  The total number of previous results to offset for this page.
</ParamField>

<ParamField query="isDropLink" type="boolean" optional>
  Optionally filter out templates for drop links.
</ParamField>

### Response

  <Expandable title="properties">

    <ResponseField name="name" type="string"/>
    <ResponseField name="animationUrl" type="string"/>
    <ResponseField name="contractId" type="string" required/>
    <ResponseField name="defaultMetadata" type="object" required/>
    <ResponseField name="delayMinting" type="boolean" required/>
    <ResponseField name="description" type="string" required/>
    <ResponseField name="gatedContent" type="array of objects" required/>
    <ResponseField name="image" type="string"/>
    <ResponseField name="isDroplink" type="boolean" required/>
    <ResponseField name="liveAt" type="date"/>
    <ResponseField name="maxTokens" type="number" required/>
    <ResponseField name="namingTemplate" type="string" required/>
    <ResponseField name="payouts" type="object" required/>
    <ResponseField name="previewAt" type="date" />
    <ResponseField name="prices" type="array of object" required/>
    <ResponseField name="published" type="boolean" required/>
    <ResponseField name="salesEndAt" type="date" />
    <ResponseField name="totalTokens" type="number" />
  </Expandable>
</ResponseField>
